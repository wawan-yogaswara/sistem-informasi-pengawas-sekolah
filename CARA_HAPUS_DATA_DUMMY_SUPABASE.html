-- HAPUS SEMUA DATA DUMMY DARI TABEL additional_tasks DELETE FROM additional_tasks WHERE name IN ( 'Pembinaan Guru', 'Rapat Koordinasi', 'Sosialisasi Program', 'Workshop Guru', 'Monitoring Pembelajaran', 'Evaluasi Kurikulum' ) OR name LIKE '%Pembinaan%' OR name LIKE '%Rapat%' OR name LIKE '%Workshop%' OR name LIKE '%Monitoring%'; -- HAPUS SEMUA DATA DUMMY DARI TABEL supervisions DELETE FROM supervisions WHERE school_name IN ( 'SDN 1 Garut', 'SDN 2 Garut', 'SMPN 1 Garut', 'SMPN 2 Garut' ) OR findings LIKE '%Pembelajaran sudah%' OR findings LIKE '%Administrasi sekolah%' OR teacher_name IN ('Ibu Siti Aminah', 'Bapak Ahmad Fauzi'); -- HAPUS SEMUA DATA DUMMY DARI TABEL activities (jika ada) DELETE FROM activities WHERE title LIKE '%Supervisi SDN%' OR title LIKE '%Penyusunan Laporan%' OR description LIKE '%supervisi akademik%'; -- HAPUS SEMUA DATA DUMMY DARI TABEL events (jika ada) DELETE FROM events WHERE name LIKE '%Supervisi SDN%' OR name LIKE '%Penyusunan Laporan%' OR description LIKE '%supervisi akademik%'; -- CEK HASIL SETELAH HAPUS SELECT 'additional_tasks' as table_name, COUNT(*) as remaining_count FROM additional_tasks UNION ALL SELECT 'supervisions' as table_name, COUNT(*) as remaining_count FROM supervisions UNION ALL SELECT 'activities' as table_name, COUNT(*) as remaining_count FROM activities UNION ALL SELECT 'events' as table_name, COUNT(*) as remaining_count FROM events;
